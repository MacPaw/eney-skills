import { Tab, Tabs } from 'fumadocs-ui/components/tabs';
import { Callout } from 'fumadocs-ui/components/callout';

All form fields are accessed as static properties on the `Form` component.

```tsx
import { Form } from "@macpaw/eney-api";

// Form.TextField, Form.Checkbox, Form.Dropdown, etc.
```

---

## TextField

Single-line text input.

![TextField widget](https://placeholder.com/widgets/text-field.png)

```tsx
<Form.TextField
  name="username"
  value={username}
  onChange={setUsername}
  label="Username"
  isCopyable
/>
```

### Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `name` | `string` | **required** | Field identifier |
| `value` | `string` | **required** | Current value |
| `onChange` | `(value: string) => void` | **required** | Change handler |
| `label` | `string` | — | Display label |
| `isCopyable` | `boolean` | — | Show a copy button next to the field |
| `$context` | `boolean` | — | Expose value as context |

## Dropdown

Select from a list of options.

![Dropdown widget](https://placeholder.com/widgets/dropdown.png)

```tsx
<Form.Dropdown
  name="language"
  value={language}
  onChange={setLanguage}
  label="Language"
  searchable
>
  <Form.Dropdown.Item value="en" title="English" />
  <Form.Dropdown.Item value="uk" title="Ukrainian" />
  <Form.Dropdown.Item value="de" title="German" />
</Form.Dropdown>
```

### Dynamic Options

```tsx
<Form.Dropdown name="note" value={selected} onChange={setSelected}>
  <Form.Dropdown.Item value="__new__" title="New Note" />
  {notes.map((note) => (
    <Form.Dropdown.Item key={note.id} value={note.title} title={note.title} />
  ))}
</Form.Dropdown>
```

### Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `name` | `string` | **required** | Field identifier |
| `value` | `string` | **required** | Currently selected value |
| `onChange` | `(value: string) => void` | **required** | Change handler |
| `label` | `string` | — | Display label |
| `searchable` | `boolean` | `false` | Enable filtering options by typing |
| `children` | `Dropdown.Item[]` | — | Available options |
| `$context` | `boolean` | — | Expose value as context |

### Dropdown.Item Props

| Prop | Type | Description |
|------|------|-------------|
| `value` | `string` | Option value |
| `title` | `string` | Display text |

## FilePicker

File selection dialog — single or multiple files.

![FilePicker widget](https://placeholder.com/widgets/file-picker.png)

<Tabs items={['Single File', 'Multiple Files']}>
<Tab value="Single File">
```tsx
<Form.FilePicker
  name="avatar"
  value={filePath}
  onChange={setFilePath}
  label="Avatar"
  accept={["image/png", "image/jpeg"]}
/>
```
</Tab>
<Tab value="Multiple Files">
```tsx
<Form.FilePicker
  name="attachments"
  multiple
  value={filePaths}
  onChange={setFilePaths}
  label="Attachments"
/>
```
</Tab>
</Tabs>

### Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `name` | `string` | **required** | Field identifier |
| `value` | `string \| string[]` | — | Selected file path(s) |
| `onChange` | `(value) => void` | **required** | Change handler (type matches `multiple`) |
| `label` | `string` | — | Display label |
| `accept` | `string[]` | — | Allowed MIME types (e.g. `["image/png"]`) |
| `multiple` | `boolean` | `false` | Allow selecting multiple files |

<Callout type="warn">
When `multiple` is `true`, `value` is `string[]` and `onChange` receives `string[]`. When `false` or omitted, both are `string`.
</Callout>
